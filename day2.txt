# import dependeny
from fastapi import FastAPI,HTTPException,status
from fastapi.params import Body
from pydantic import BaseModel
from random import randrange

app= FastAPI()
# dummy data
my_post=[{"title":"welcome title","content":"dummy content","id":101}]

# search function

def findPost(id):
    for i,p in enumerate(my_post):
        if p['id'] == id:
            return i

# Declaring Model

class Post(BaseModel):
    title: str
    content: str

# load the data using get

@app.get("/loadposts")
def loadallposts():
    return {"dummy-message": my_post}

# get specific record by passing ID
@app.get("/loadpost/{id}")
def loadById(id:int):
    post=findPost(id)
    if not post:
        raise HTTPException(status_code=status.HTTP_404_NOT_FOUND,detail='Given ID not found')
    else:
    
        return {"User Found":post}
    
# delete record by passing ID
@app.delete("/deletepost/{id}")
def deleteById(id:int):
    post=findPost(id)
    if not post:
        raise HTTPException(status_code=status.HTTP_404_NOT_FOUND,detail='Given ID not found')
    my_post.pop(post)
    return {"message": "user deleted"}

# add the new post

@app.post("/addpost",status_code=status.HTTP_201_CREATED)
def addpost(payload: Post):
    data= payload.dict()
    data['id']=randrange(0,1000)
    my_post.append(data)
    return {"message":data}
